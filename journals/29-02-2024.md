local function getPreviousday(day)
    local parsedDate = os.time {
        year = tonumber(day[3]),
        month = tonumber(day[2]),
        day = tonumber(day[1])
    }
    local previousDayTime = parsedDate - (24 * 60 * 60)
    local formatedDate = os.date("%d-%m-%Y", previousDayTime)
    return splitString(formatedDate, '-')
end
